One-time setup

Projects table missing
Run this once in Supabase SQL Editor (project baazubvfsrpmbfmrzumw) and this page goes fully live with your real projects.
create table if not exists projects (
  id uuid primary key default gen_random_uuid(),
  name text not null, owner text,
  progress int default 0,
  status text default 'on-track',
  deadline date,
  tasks_done int default 0, tasks_total int default 0,
  created_at timestamp default now());
alter table projects enable row level security;
drop policy if exists "allow all projects" on projects;
create policy "allow all projects" on projects for all using (true) with check (true);
insert into projects (name,owner,progress,status,tasks_done,tasks_total) values
 ('Website Redesign','Amjad',75,'on-track',12,18),
 ('MTI Course Development','Sreekanth',60,'on-track',18,28),
 ('Content Creation Pipeline','Anal',80,'on-track',14,16),
 ('Offline Center Setup','Hasnu',30,'at-risk',6,20);
Then edit rows any time in Supabase → Table Editor → projects.
